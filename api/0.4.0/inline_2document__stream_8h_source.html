21 inline void stage1_worker::start_thread() {
22  std::unique_lock<std::mutex> lock(locking_mutex);
23  if(thread.joinable()) {
24  return; // This should never happen but we never want to create more than one thread.
25  }
26  thread = std::thread([this]{
27  while(can_work) {
28  std::unique_lock<std::mutex> thread_lock(locking_mutex);
29  cond_var.wait(thread_lock, [this]{return has_work || !can_work;});
30  if(!can_work) {
31  break;
32  }
33  this->owner->stage1_thread_error = this->owner->run_stage1(*this->stage1_thread_parser,
34  this->_next_batch_start);
35  this->has_work = false;
36  thread_lock.unlock();
37  cond_var.notify_one(); // will notify "finish"
38  }
39  }
40  );
41 }

128 inline void document_stream::start() noexcept {
129  if (error) { return; }
130 
131  error = parser->ensure_capacity(batch_size);
132  if (error) { return; }
133 
134  // Always run the first stage 1 parse immediately
135  batch_start = 0;
136  error = run_stage1(*parser, batch_start);
137  if (error) { return; }
138 
139 #ifdef SIMDJSON_THREADS_ENABLED
140  if (next_batch_start() < len) {

143  if (error) { return; }
144  worker->start_thread();
145  start_stage1_thread();
146  if (error) { return; }
147  }
148 #endif // SIMDJSON_THREADS_ENABLED
149 
150  next();
151 }

156 inline void document_stream::next() noexcept {
157  if (error) { return; }
158 
159  // Load the next document from the batch
160  doc_index = batch_start + parser->implementation->structural_indexes[parser->implementation->next_structural_index];
161  error = parser->implementation->stage2_next(parser->doc);
162  // If that was the last document in the batch, load another batch (if available)
163  while (error == EMPTY) {
164  batch_start = next_batch_start();
165  if (batch_start >= len) { break; }
166 
167 #ifdef SIMDJSON_THREADS_ENABLED
168  load_from_stage1_thread();
169 #else
170  error = run_stage1(*parser, batch_start);
171 #endif
172  if (error) { continue; } // If the error was EMPTY, we may want to load another batch.
173  // Run stage 2 on the first document in the batch
174  doc_index = batch_start + parser->implementation->structural_indexes[parser->implementation->next_structural_index];
175  error = parser->implementation->stage2_next(parser->doc);
176  }
177 }

195 inline void document_stream::load_from_stage1_thread() noexcept {
196  worker->finish();
197  // Swap to the parser that was loaded up in the thread. Make sure the parser has
198  // enough memory to swap to, as well.
199  std::swap(*parser, stage1_thread_parser);
200  error = stage1_thread_error;
201  if (error) { return; }
202 
203  // If there's anything left, start the stage 1 thread!

206  }

210  // we call the thread on a lambda that will update
