\begin{DoxyCode}{0}
\DoxyCodeLine{set(SIMDJSON\_JUST\_LIBRARY ON CACHE STRING "Build just the library, nothing else." FORCE)}
\DoxyCodeLine{add\_subdirectory(simdjson EXCLUDE\_FROM\_ALL)}
\end{DoxyCode}
