107 simdjson_inline simdjson_result<size_t> value::count_elements() & noexcept {
108  simdjson_result<size_t> answer;
109  auto a = get_array();
110  answer = a.count_elements();
111  // count_elements leaves you pointing inside the array, at the first element.
112  // We need to move back so that the user can create a new array (which requires that
113  // we point at '[').
114  iter.move_at_start();
115  return answer;

187 simdjson_inline simdjson_result<value> value::at_pointer(std::string_view json_pointer) noexcept {
188  json_type t;
189  SIMDJSON_TRY(type().get(t));
190  switch (t)
191  {
192  case json_type::array:
193  return (*this).get_array().at_pointer(json_pointer);
194  case json_type::object:
195  return (*this).get_object().at_pointer(json_pointer);
196  default:
197  return INVALID_JSON_POINTER;
198  }
199 }
