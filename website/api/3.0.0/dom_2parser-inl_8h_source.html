33 inline simdjson_result<size_t> parser::read_file(const std::string &path) noexcept {
34  // Open the file
35  SIMDJSON_PUSH_DISABLE_WARNINGS
36  SIMDJSON_DISABLE_DEPRECATED_WARNING // Disable CRT_SECURE warning on MSVC: manually verified this is safe
37  std::FILE *fp = std::fopen(path.c_str(), "rb");
38  SIMDJSON_POP_DISABLE_WARNINGS
39 
40  if (fp == nullptr) {
41  return IO_ERROR;
42  }
43 
44  // Get the file size
45  int ret;
46 #if defined(SIMDJSON_VISUAL_STUDIO) && !SIMDJSON_IS_32BITS
47  ret = _fseeki64(fp, 0, SEEK_END);
48 #else
49  ret = std::fseek(fp, 0, SEEK_END);
50 #endif // _WIN64
51  if(ret < 0) {
52  std::fclose(fp);
53  return IO_ERROR;
54  }
55 #if defined(SIMDJSON_VISUAL_STUDIO) && !SIMDJSON_IS_32BITS
56  __int64 len = _ftelli64(fp);
57  if(len == -1L) {
58  std::fclose(fp);
59  return IO_ERROR;
60  }
61 #else
62  long len = std::ftell(fp);
63  if((len < 0) || (len == LONG_MAX)) {
64  std::fclose(fp);
65  return IO_ERROR;
66  }
67 #endif
68 
69  // Make sure we have enough capacity to load the file
70  if (_loaded_bytes_capacity < size_t(len)) {
71  loaded_bytes.reset( internal::allocate_padded_buffer(len) );
72  if (!loaded_bytes) {
73  std::fclose(fp);
74  return MEMALLOC;
75  }
76  _loaded_bytes_capacity = len;
77  }
78 
79  // Read the string
80  std::rewind(fp);
81  size_t bytes_read = std::fread(loaded_bytes.get(), 1, len, fp);
82  if (std::fclose(fp) != 0 || bytes_read != size_t(len)) {
83  return IO_ERROR;
84  }
85 
86  return bytes_read;
87 }

104 inline simdjson_result<element> parser::parse_into_document(document& provided_doc, const uint8_t *buf, size_t len, bool realloc_if_needed) & noexcept {
105  // Important: we need to ensure that document has enough capacity.
106  // Important: It is possible that provided_doc is actually the internal 'doc' within the parser!!!
107  error_code _error = ensure_capacity(provided_doc, len);
108  if (_error) { return _error; }
109  if (realloc_if_needed) {
110  // Make sure we have enough capacity to copy len bytes
111  if (!loaded_bytes || _loaded_bytes_capacity < len) {
112  loaded_bytes.reset( internal::allocate_padded_buffer(len) );
113  if (!loaded_bytes) {
114  return MEMALLOC;
115  }
116  _loaded_bytes_capacity = len;
117  }
118  std::memcpy(static_cast<void *>(loaded_bytes.get()), buf, len);
119  }
120  _error = implementation->parse(realloc_if_needed ? reinterpret_cast<const uint8_t*>(loaded_bytes.get()): buf, len, provided_doc);
121 
122  if (_error) { return _error; }
123 
124  return provided_doc.root();
125 }

203 inline error_code parser::ensure_capacity(document& target_document, size_t desired_capacity) noexcept {
204  // 1. It is wasteful to allocate a document and a parser for documents spanning less than MINIMAL_DOCUMENT_CAPACITY bytes.
205  // 2. If we allow desired_capacity = 0 then it is possible to exit this function with implementation == nullptr.
206  if(desired_capacity < MINIMAL_DOCUMENT_CAPACITY) { desired_capacity = MINIMAL_DOCUMENT_CAPACITY; }
207  // If we don't have enough capacity, (try to) automatically bump it.
208  // If the document needs allocation, do it too.
209  // Both in one if statement to minimize unlikely branching.
210  //
211  // Note: we must make sure that this function is called if capacity() == 0. We do so because we
212  // ensure that desired_capacity > 0.
213  if (simdjson_unlikely(capacity() < desired_capacity || target_document.capacity() < desired_capacity)) {
214  if (desired_capacity > max_capacity()) {
215  return error = CAPACITY;
216  }
217  error_code err1 = target_document.capacity() < desired_capacity ? target_document.allocate(desired_capacity) : SUCCESS;
218  error_code err2 = capacity() < desired_capacity ? allocate(desired_capacity, max_depth()) : SUCCESS;
219  if(err1 != SUCCESS) { return error = err1; }
220  if(err2 != SUCCESS) { return error = err2; }
221  }
222  return SUCCESS;
223 }
