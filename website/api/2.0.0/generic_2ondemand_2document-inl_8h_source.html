47 simdjson_really_inline simdjson_result<value> document::get_value() noexcept {
48  // Make sure we start any arrays or objects before returning, so that start_root_<object/array>()
49  // gets called.
50  iter.assert_at_document_depth();
51  switch (*iter.peek()) {
52  case '[':
53  case '{':
54  return value(get_root_value_iterator());
55  default:
56  // Unfortunately, scalar documents are a special case in simdjson and they cannot
57  // be safely converted to value instances.
58  return SCALAR_DOCUMENT_AS_VALUE;
59  // return value(get_root_value_iterator());
60  }
61 }

140  simdjson_result<size_t> answer = a.count_elements();

144  iter.assert_at_document_depth();

194 simdjson_really_inline simdjson_result<std::string_view> document::raw_json() noexcept {
195  auto _iter = get_root_value_iterator();
196  const uint8_t * starting_point{_iter.peek_start()};
197  auto error = consume();
198  if(error) { return error; }
199  // After 'consume()', we could be left pointing just beyond the document, but that
200  // is ok because we are not going to dereference the final pointer position, we just
201  // use it to compute the length in bytes.
202  const uint8_t * final_point{iter.unsafe_pointer()};
203  return std::string_view(reinterpret_cast<const char*>(starting_point), size_t(final_point - starting_point));
204 }

239 simdjson_really_inline simdjson_result<value> document::at_pointer(std::string_view json_pointer) noexcept {
240  rewind(); // Rewind the document each time at_pointer is called
241  if (json_pointer.empty()) {
242  return this->get_value();
243  }
244  json_type t;
245  SIMDJSON_TRY(type().get(t));
246  switch (t)
247  {
248  case json_type::array:
249  return (*this).get_array().at_pointer(json_pointer);
250  case json_type::object:
251  return (*this).get_object().at_pointer(json_pointer);
252  default:
253  return INVALID_JSON_POINTER;
254  }
255 }
