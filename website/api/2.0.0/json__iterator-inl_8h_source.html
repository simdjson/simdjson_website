55 simdjson_warn_unused simdjson_really_inline error_code json_iterator::skip_child(depth_t parent_depth) noexcept {
56  if (depth() <= parent_depth) { return SUCCESS; }
57  switch (*return_current_and_advance()) {
58  // TODO consider whether matching braces is a requirement: if non-matching braces indicates
59  // *missing* braces, then future lookups are not in the object/arrays they think they are,
60  // violating the rule "validate enough structure that the user can be confident they are
61  // looking at the right values."
62  // PERF TODO we can eliminate the switch here with a lookup of how much to add to depth
63 
64  // For the first open array/object in a value, we've already incremented depth, so keep it the same
65  // We never stop at colon, but if we did, it wouldn't affect depth
66  case '[': case '{': case ':':
67  logger::log_start_value(*this, "skip");
68  break;
69  // If there is a comma, we have just finished a value in an array/object, and need to get back in
70  case ',':
71  logger::log_value(*this, "skip");
72  break;
73  // ] or } means we just finished a value and need to jump out of the array/object
74  case ']': case '}':
75  logger::log_end_value(*this, "skip");
76  _depth--;
77  if (depth() <= parent_depth) { return SUCCESS; }
78 #if SIMDJSON_CHECK_EOF
79  // If there are no more tokens, the parent is incomplete.
80  if (at_end()) { return report_error(INCOMPLETE_ARRAY_OR_OBJECT, "Missing [ or { at start"); }
81 #endif // SIMDJSON_CHECK_EOF
82  break;
83  case '"':
84  if(*peek() == ':') {
85  // We are at a key!!!
86  // This might happen if you just started an object and you skip it immediately.
87  // Performance note: it would be nice to get rid of this check as it is somewhat
88  // expensive.
89  // https://github.com/simdjson/simdjson/issues/1742
90  logger::log_value(*this, "key");
91  return_current_and_advance(); // eat up the ':'
92  break; // important!!!
93  }
94  simdjson_fallthrough;
95  // Anything else must be a scalar value
96  default:
97  // For the first scalar, we will have incremented depth already, so we decrement it here.
98  logger::log_value(*this, "skip");
99  _depth--;
100  if (depth() <= parent_depth) { return SUCCESS; }
101  break;
102  }
103 
104  // Now that we've considered the first value, we only increment/decrement for arrays/objects
105  while (position() < end_position()) {
106  switch (*return_current_and_advance()) {
107  case '[': case '{':
108  logger::log_start_value(*this, "skip");
109  _depth++;
110  break;
111  // TODO consider whether matching braces is a requirement: if non-matching braces indicates
112  // *missing* braces, then future lookups are not in the object/arrays they think they are,
113  // violating the rule "validate enough structure that the user can be confident they are
114  // looking at the right values."
115  // PERF TODO we can eliminate the switch here with a lookup of how much to add to depth
116  case ']': case '}':
117  logger::log_end_value(*this, "skip");
118  _depth--;
119  if (depth() <= parent_depth) { return SUCCESS; }
120  break;
121  default:
122  logger::log_value(*this, "skip", "");
123  break;
124  }
125  }
126 
127  return report_error(TAPE_ERROR, "not enough close braces");
128 }
