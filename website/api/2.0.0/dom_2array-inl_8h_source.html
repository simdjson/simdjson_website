67 inline simdjson_result<element> array::at_pointer(std::string_view json_pointer) const noexcept {
68  if(json_pointer.empty()) { // an empty string means that we return the current node
69  return element(this->tape); // copy the current node
70  } else if(json_pointer[0] != '/') { // otherwise there is an error
71  return INVALID_JSON_POINTER;
72  }
73  json_pointer = json_pointer.substr(1);
74  // - means "the append position" or "the element after the end of the array"
75  // We don't support this, because we're returning a real element, not a position.
76  if (json_pointer == "-") { return INDEX_OUT_OF_BOUNDS; }
77 
78  // Read the array index
79  size_t array_index = 0;
80  size_t i;
81  for (i = 0; i < json_pointer.length() && json_pointer[i] != '/'; i++) {
82  uint8_t digit = uint8_t(json_pointer[i] - '0');
83  // Check for non-digit in array index. If it's there, we're trying to get a field in an object
84  if (digit > 9) { return INCORRECT_TYPE; }
85  array_index = array_index*10 + digit;
86  }
87 
88  // 0 followed by other digits is invalid
89  if (i > 1 && json_pointer[0] == '0') { return INVALID_JSON_POINTER; } // "JSON pointer array index has other characters after 0"
90 
91  // Empty string is invalid; so is a "/" with no digits before it
92  if (i == 0) { return INVALID_JSON_POINTER; } // "Empty string in JSON pointer array index"
93 
94  // Get the child
95  auto child = array(tape).at(array_index);
96  // If there is an error, it ends here
97  if(child.error()) {
98  return child;
99  }
100  // If there is a /, we're not done yet, call recursively.
101  if (i < json_pointer.length()) {
102  child = child.at_pointer(json_pointer.substr(i));
103  }
104  return child;
105 }
