83 inline simdjson_result<element> object::at(const std::string_view &json_pointer) const noexcept {
84  size_t slash = json_pointer.find('/');
85  std::string_view key = json_pointer.substr(0, slash);
86 
87  // Grab the child with the given key
88  simdjson_result<element> child;
89 
90  // If there is an escape character in the key, unescape it and then get the child.
91  size_t escape = key.find('~');
92  if (escape != std::string_view::npos) {
93  // Unescape the key
94  std::string unescaped(key);
95  do {
96  switch (unescaped[escape+1]) {
97  case '0':
98  unescaped.replace(escape, 2, "~");
99  break;
100  case '1':
101  unescaped.replace(escape, 2, "/");
102  break;
103  default:
104  return INVALID_JSON_POINTER; // "Unexpected ~ escape character in JSON pointer");
105  }
106  escape = unescaped.find('~', escape+1);
107  } while (escape != std::string::npos);
108  child = at_key(unescaped);
109  } else {
110  child = at_key(key);
111  }
112 
113  // If there is a /, we have to recurse and look up more of the path
114  if (slash != std::string_view::npos) {
115  child = child.at(json_pointer.substr(slash+1));
116  }
117 
118  return child;
119 }

183 inline bool object::iterator::key_equals(const std::string_view & o) const noexcept {
184  // We use the fact that the key length can be computed quickly
185  // without access to the string buffer.
186  const uint32_t len = key_length();
187  if(o.size() == len) {
188  // We avoid construction of a temporary string_view instance.
189  return (memcmp(o.data(), key_c_str(), len) == 0);
190  }
191  return false;

194 inline bool object::iterator::key_equals_case_insensitive(const std::string_view & o) const noexcept {
195  // We use the fact that the key length can be computed quickly
196  // without access to the string buffer.
197  const uint32_t len = key_length();
198  if(o.size() == len) {
199  // See For case-insensitive string comparisons, avoid char-by-char functions
200  // https://lemire.me/blog/2020/04/30/for-case-insensitive-string-comparisons-avoid-char-by-char-functions/
201  // Note that it might be worth rolling our own strncasecmp function, with vectorization.
202  return (simdjson_strncasecmp(o.data(), key_c_str(), len) == 0);
203  }
204  return false;
205 }
