37 inline simdjson_result<size_t> parser::read_file(const std::string &path) noexcept {
38  // Open the file
39  SIMDJSON_PUSH_DISABLE_WARNINGS
40  SIMDJSON_DISABLE_DEPRECATED_WARNING // Disable CRT_SECURE warning on MSVC: manually verified this is safe
41  std::FILE *fp = std::fopen(path.c_str(), "rb");
42  SIMDJSON_POP_DISABLE_WARNINGS
43 
44  if (fp == nullptr) {
45  return IO_ERROR;
46  }
47 
48  // Get the file size
49  if(std::fseek(fp, 0, SEEK_END) < 0) {
50  std::fclose(fp);
51  return IO_ERROR;
52  }
53  long len = std::ftell(fp);
54  if((len < 0) || (len == LONG_MAX)) {
55  std::fclose(fp);
56  return IO_ERROR;
57  }
58 
59  // Make sure we have enough capacity to load the file
60  if (_loaded_bytes_capacity < size_t(len)) {
61  loaded_bytes.reset( internal::allocate_padded_buffer(len) );
62  if (!loaded_bytes) {
63  std::fclose(fp);
64  return MEMALLOC;
65  }
66  _loaded_bytes_capacity = len;
67  }
68 
69  // Read the string
70  std::rewind(fp);
71  size_t bytes_read = std::fread(loaded_bytes.get(), 1, len, fp);
72  if (std::fclose(fp) != 0 || bytes_read != size_t(len)) {
73  return IO_ERROR;
74  }
75 
76  return bytes_read;
77 }

94  error_code _error = ensure_capacity(len);
95  if (_error) { return _error; }
96 
97  if (realloc_if_needed) {
98  const uint8_t *tmp_buf = buf;
99  buf = (uint8_t *)internal::allocate_padded_buffer(len);
100  if (buf == nullptr)
101  return MEMALLOC;

108  }
109  if (_error) { return _error; }

147 inline error_code parser::allocate(size_t capacity, size_t max_depth) noexcept {
148  //
149  // Reallocate implementation and document if needed
150  //
151  error_code err;
152  //
153  // It is possible that we change max_depth without touching capacity, in
154  // which case, we do not want to reallocate the document buffers.
155  //
156  bool need_doc_allocation{false};
157  if (implementation) {
158  need_doc_allocation = implementation->capacity() != capacity || !doc.tape;
159  err = implementation->allocate(capacity, max_depth);
160  } else {
161  need_doc_allocation = true;
162  err = simdjson::active_implementation->create_dom_parser_implementation(capacity, max_depth, implementation);
163  }
164  if (err) { return err; }
165  if (need_doc_allocation) {
166  err = doc.allocate(capacity);
167  if (err) { return err; }
168  }
169  return SUCCESS;
170 }

177 inline error_code parser::ensure_capacity(size_t desired_capacity) noexcept {
178  // If we don't have enough capacity, (try to) automatically bump it.
179  // If the document was taken, reallocate that too.
180  // Both in one if statement to minimize unlikely branching.
181  if (unlikely(capacity() < desired_capacity || !doc.tape)) {
182  if (desired_capacity > max_capacity()) {
183  return error = CAPACITY;
184  }
185  return allocate(desired_capacity, max_depth());
186  }
187 
188  return SUCCESS;
189 }
