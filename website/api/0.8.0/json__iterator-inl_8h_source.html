37 simdjson_warn_unused simdjson_really_inline error_code json_iterator::skip_child(depth_t parent_depth) noexcept {
38  if (depth() <= parent_depth) { return SUCCESS; }
39 
40  switch (*advance()) {
41  // TODO consider whether matching braces is a requirement: if non-matching braces indicates
42  // *missing* braces, then future lookups are not in the object/arrays they think they are,
43  // violating the rule "validate enough structure that the user can be confident they are
44  // looking at the right values."
45  // PERF TODO we can eliminate the switch here with a lookup of how much to add to depth
46 
47  // For the first open array/object in a value, we've already incremented depth, so keep it the same
48  // We never stop at colon, but if we did, it wouldn't affect depth
49  case '[': case '{': case ':':
50  logger::log_start_value(*this, "skip");
51  break;
52  // If there is a comma, we have just finished a value in an array/object, and need to get back in
53  case ',':
54  logger::log_value(*this, "skip");
55  break;
56  // ] or } means we just finished a value and need to jump out of the array/object
57  case ']': case '}':
58  logger::log_end_value(*this, "skip");
59  _depth--;
60  if (depth() <= parent_depth) { return SUCCESS; }
61  break;
62  // Anything else must be a scalar value
63  default:
64  // For the first scalar, we will have incremented depth already, so we decrement it here.
65  logger::log_value(*this, "skip");
66  _depth--;
67  if (depth() <= parent_depth) { return SUCCESS; }
68  break;
69  }
70 
71  // Now that we've considered the first value, we only increment/decrement for arrays/objects
72  auto end = &parser->dom_parser->structural_indexes[parser->dom_parser->n_structural_indexes];
73  while (token.index <= end) {
74  switch (*advance()) {
75  case '[': case '{':
76  logger::log_start_value(*this, "skip");
77  _depth++;
78  break;
79  // TODO consider whether matching braces is a requirement: if non-matching braces indicates
80  // *missing* braces, then future lookups are not in the object/arrays they think they are,
81  // violating the rule "validate enough structure that the user can be confident they are
82  // looking at the right values."
83  // PERF TODO we can eliminate the switch here with a lookup of how much to add to depth
84  case ']': case '}':
85  logger::log_end_value(*this, "skip");
86  _depth--;
87  if (depth() <= parent_depth) { return SUCCESS; }
88  break;
89  default:
90  logger::log_value(*this, "skip", "");
91  break;
92  }
93  }
94 
95  return report_error(TAPE_ERROR, "not enough close braces");
96 }
