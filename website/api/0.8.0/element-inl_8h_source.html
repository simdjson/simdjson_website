227 inline simdjson_result<uint64_t> element::get_uint64() const noexcept {
228  if(simdjson_unlikely(!tape.is_uint64())) { // branch rarely taken
229  if(tape.is_int64()) {
230  int64_t result = tape.next_tape_value<int64_t>();
231  if (result < 0) {
232  return NUMBER_OUT_OF_RANGE;
233  }
234  return uint64_t(result);
235  }
236  return INCORRECT_TYPE;
237  }
238  return tape.next_tape_value<int64_t>();
239 }
240 inline simdjson_result<int64_t> element::get_int64() const noexcept {
241  if(simdjson_unlikely(!tape.is_int64())) { // branch rarely taken
242  if(tape.is_uint64()) {
243  uint64_t result = tape.next_tape_value<uint64_t>();
244  // Wrapping max in parens to handle Windows issue: https://stackoverflow.com/questions/11544073/how-do-i-deal-with-the-max-macro-in-windows-h-colliding-with-max-in-std
245  if (result > uint64_t((std::numeric_limits<int64_t>::max)())) {
246  return NUMBER_OUT_OF_RANGE;
247  }
248  return static_cast<int64_t>(result);
249  }
250  return INCORRECT_TYPE;
251  }
252  return tape.next_tape_value<int64_t>();
253 }
254 inline simdjson_result<double> element::get_double() const noexcept {
255  // Performance considerations:
256  // 1. Querying tape_ref_type() implies doing a shift, it is fast to just do a straight
257  // comparison.
258  // 2. Using a switch-case relies on the compiler guessing what kind of code generation
259  // we want... But the compiler cannot know that we expect the type to be "double"
260  // most of the time.
261  // We can expect get<double> to refer to a double type almost all the time.
262  // It is important to craft the code accordingly so that the compiler can use this
263  // information. (This could also be solved with profile-guided optimization.)
264  if(simdjson_unlikely(!tape.is_double())) { // branch rarely taken
265  if(tape.is_uint64()) {
266  return double(tape.next_tape_value<uint64_t>());
267  } else if(tape.is_int64()) {
268  return double(tape.next_tape_value<int64_t>());
269  }
270  return INCORRECT_TYPE;
271  }
272  // this is common:
273  return tape.next_tape_value<double>();
274 }

362 inline simdjson_result<element> element::at_pointer(std::string_view json_pointer) const noexcept {
363  switch (tape.tape_ref_type()) {
364  case internal::tape_type::START_OBJECT:
365  return object(tape).at_pointer(json_pointer);
366  case internal::tape_type::START_ARRAY:
367  return array(tape).at_pointer(json_pointer);
368  default: {
369  if(!json_pointer.empty()) { // a non-empty string is invalid on an atom
370  return INVALID_JSON_POINTER;
371  }
372  // an empty string means that we return the current node
373  dom::element copy(*this);
374  return simdjson_result<element>(std::move(copy));
375  }
376  }
