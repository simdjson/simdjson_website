36 // handle a unicode codepoint
37 // write appropriate values into dest
38 // src will advance 6 bytes or 12 bytes
39 // dest will advance a variable amount (return via pointer)
40 // return true if the unicode codepoint was valid
41 // We work in little-endian then swap at write time
42 simdjson_warn_unused
43 simdjson_really_inline bool handle_unicode_codepoint(const uint8_t **src_ptr,
44  uint8_t **dst_ptr) {
45  // jsoncharutils::hex_to_u32_nocheck fills high 16 bits of the return value with 1s if the
46  // conversion isn't valid; we defer the check for this to inside the
47  // multilingual plane check
48  uint32_t code_point = jsoncharutils::hex_to_u32_nocheck(*src_ptr + 2);
49  *src_ptr += 6;
50  // check for low surrogate for characters outside the Basic
51  // Multilingual Plane.
52  if (code_point >= 0xd800 && code_point < 0xdc00) {
53  if (((*src_ptr)[0] != '\\') || (*src_ptr)[1] != 'u') {
54  return false;
55  }
56  uint32_t code_point_2 = jsoncharutils::hex_to_u32_nocheck(*src_ptr + 2);
57 
58  // if the first code point is invalid we will get here, as we will go past
59  // the check for being outside the Basic Multilingual plane. If we don't
60  // find a \u immediately afterwards we fail out anyhow, but if we do,
61  // this check catches both the case of the first code point being invalid
62  // or the second code point being invalid.
63  if ((code_point | code_point_2) >> 16) {
64  return false;
65  }
66 
67  code_point =
68  (((code_point - 0xd800) << 10) | (code_point_2 - 0xdc00)) + 0x10000;
69  *src_ptr += 6;
70  }
71  size_t offset = jsoncharutils::codepoint_to_utf8(code_point, *dst_ptr);
72  *dst_ptr += offset;
73  return offset > 0;
74 }

80 simdjson_warn_unused simdjson_really_inline uint8_t *parse_string(const uint8_t *src, uint8_t *dst) {
81  while (1) {
82  // Copy the next n bytes, and find the backslash and quote in them.
83  auto bs_quote = backslash_and_quote::copy_and_find(src, dst);
84  // If the next thing is the end quote, copy and return
85  if (bs_quote.has_quote_first()) {
86  // we encountered quotes first. Move dst to point to quotes and exit
87  return dst + bs_quote.quote_index();
88  }
89  if (bs_quote.has_backslash()) {
90  /* find out where the backspace is */
91  auto bs_dist = bs_quote.backslash_index();
92  uint8_t escape_char = src[bs_dist + 1];
93  /* we encountered backslash first. Handle backslash */
94  if (escape_char == 'u') {
95  /* move src/dst up to the start; they will be further adjusted
96  within the unicode codepoint handling code. */
97  src += bs_dist;
98  dst += bs_dist;
99  if (!handle_unicode_codepoint(&src, &dst)) {
100  return nullptr;
101  }
102  } else {
103  /* simple 1:1 conversion. Will eat bs_dist+2 characters in input and
104  * write bs_dist+1 characters to output
105  * note this may reach beyond the part of the buffer we've actually
106  * seen. I think this is ok */
107  uint8_t escape_result = escape_map[escape_char];
108  if (escape_result == 0u) {
109  return nullptr; /* bogus escape value is an error */
110  }
111  dst[bs_dist] = escape_result;
112  src += bs_dist + 2;
113  dst += bs_dist + 1;
114  }
115  } else {
116  /* they are the same. Since they can't co-occur, it means we
117  * encountered neither. */
118  src += backslash_and_quote::BYTES_PROCESSED;
119  dst += backslash_and_quote::BYTES_PROCESSED;
120  }
121  }
122  /* can't be reached */
123  return nullptr;
124 }
