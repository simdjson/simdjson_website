33 inline simdjson_result<size_t> parser::read_file(const std::string &path) noexcept {
34  // Open the file
35  SIMDJSON_PUSH_DISABLE_WARNINGS
36  SIMDJSON_DISABLE_DEPRECATED_WARNING // Disable CRT_SECURE warning on MSVC: manually verified this is safe
37  std::FILE *fp = std::fopen(path.c_str(), "rb");
38  SIMDJSON_POP_DISABLE_WARNINGS
39 
40  if (fp == nullptr) {
41  return IO_ERROR;
42  }
43 
44  // Get the file size
45  if(std::fseek(fp, 0, SEEK_END) < 0) {
46  std::fclose(fp);
47  return IO_ERROR;
48  }
49 #if defined(SIMDJSON_VISUAL_STUDIO) && !SIMDJSON_IS_32BITS
50  __int64 len = _ftelli64(fp);
51  if(len == -1L) {
52  std::fclose(fp);
53  return IO_ERROR;
54  }
55 #else
56  long len = std::ftell(fp);
57  if((len < 0) || (len == LONG_MAX)) {
58  std::fclose(fp);
59  return IO_ERROR;
60  }
61 #endif
62 
63  // Make sure we have enough capacity to load the file
64  if (_loaded_bytes_capacity < size_t(len)) {
65  loaded_bytes.reset( internal::allocate_padded_buffer(len) );
66  if (!loaded_bytes) {
67  std::fclose(fp);
68  return MEMALLOC;
69  }
70  _loaded_bytes_capacity = len;
71  }
72 
73  // Read the string
74  std::rewind(fp);
75  size_t bytes_read = std::fread(loaded_bytes.get(), 1, len, fp);
76  if (std::fclose(fp) != 0 || bytes_read != size_t(len)) {
77  return IO_ERROR;
78  }
79 
80  return bytes_read;
81 }

99  // Important: we need to ensure that document has enough capacity.
100  // Important: It is possible that provided_doc is actually the internal 'doc' within the parser!!!
101  error_code _error = ensure_capacity(provided_doc, len);
102  if (_error) { return _error; }
103  std::unique_ptr<uint8_t[]> tmp_buf;
104  if (realloc_if_needed) {
105  tmp_buf.reset(reinterpret_cast<uint8_t *>( internal::allocate_padded_buffer(len) ));
106  if (tmp_buf.get() == nullptr) { return MEMALLOC; }
107  std::memcpy(static_cast<void *>(tmp_buf.get()), buf, len);
108  }
109  _error = implementation->parse(realloc_if_needed ? tmp_buf.get() : buf, len, provided_doc);
110 
111  if (_error) { return _error; }
112 
113  return provided_doc.root();

192 inline error_code parser::ensure_capacity(document& target_document, size_t desired_capacity) noexcept {
193  // 1. It is wasteful to allocate a document and a parser for documents spanning less than MINIMAL_DOCUMENT_CAPACITY bytes.
194  // 2. If we allow desired_capacity = 0 then it is possible to exit this function with implementation == nullptr.
195  if(desired_capacity < MINIMAL_DOCUMENT_CAPACITY) { desired_capacity = MINIMAL_DOCUMENT_CAPACITY; }
196  // If we don't have enough capacity, (try to) automatically bump it.
197  // If the document needs allocation, do it too.
198  // Both in one if statement to minimize unlikely branching.
199  //
200  // Note: we must make sure that this function is called if capacity() == 0. We do so because we
201  // ensure that desired_capacity > 0.
202  if (simdjson_unlikely(capacity() < desired_capacity || target_document.capacity() < desired_capacity)) {
203  if (desired_capacity > max_capacity()) {
204  return error = CAPACITY;
205  }
206  error_code err1 = target_document.capacity() < desired_capacity ? target_document.allocate(desired_capacity) : SUCCESS;
207  error_code err2 = capacity() < desired_capacity ? allocate(desired_capacity, max_depth()) : SUCCESS;
208  if(err1 != SUCCESS) { return error = err1; }
209  if(err2 != SUCCESS) { return error = err2; }
210  }
211  return SUCCESS;
212 }
