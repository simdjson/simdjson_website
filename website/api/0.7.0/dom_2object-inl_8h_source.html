83 inline simdjson_result<element> object::at_pointer(std::string_view json_pointer) const noexcept {
84  if(json_pointer.empty()) { // an empty string means that we return the current node
85  return element(this->tape); // copy the current node
86  } else if(json_pointer[0] != '/') { // otherwise there is an error
87  return INVALID_JSON_POINTER;
88  }
89  json_pointer = json_pointer.substr(1);
90  size_t slash = json_pointer.find('/');
91  std::string_view key = json_pointer.substr(0, slash);
92  // Grab the child with the given key
93  simdjson_result<element> child;
94 
95  // If there is an escape character in the key, unescape it and then get the child.
96  size_t escape = key.find('~');
97  if (escape != std::string_view::npos) {
98  // Unescape the key
99  std::string unescaped(key);
100  do {
101  switch (unescaped[escape+1]) {
102  case '0':
103  unescaped.replace(escape, 2, "~");
104  break;
105  case '1':
106  unescaped.replace(escape, 2, "/");
107  break;
108  default:
109  return INVALID_JSON_POINTER; // "Unexpected ~ escape character in JSON pointer");
110  }
111  escape = unescaped.find('~', escape+1);
112  } while (escape != std::string::npos);
113  child = at_key(unescaped);
114  } else {
115  child = at_key(key);
116  }
117  if(child.error()) {
118  return child; // we do not continue if there was an error
119  }
120  // If there is a /, we have to recurse and look up more of the path
121  if (slash != std::string_view::npos) {
122  child = child.at_pointer(json_pointer.substr(slash));
123  }
124  return child;
125 }

210 inline bool object::iterator::key_equals(std::string_view o) const noexcept {
211  // We use the fact that the key length can be computed quickly
212  // without access to the string buffer.
213  const uint32_t len = key_length();

221 inline bool object::iterator::key_equals_case_insensitive(std::string_view o) const noexcept {
222  // We use the fact that the key length can be computed quickly
223  // without access to the string buffer.
224  const uint32_t len = key_length();
225  if(o.size() == len) {
226  // See For case-insensitive string comparisons, avoid char-by-char functions
227  // https://lemire.me/blog/2020/04/30/for-case-insensitive-string-comparisons-avoid-char-by-char-functions/
228  // Note that it might be worth rolling our own strncasecmp function, with vectorization.
229  return (simdjson_strncasecmp(o.data(), key_c_str(), len) == 0);
230  }
231  return false;
232 }
