24 inline error_code document::allocate(size_t capacity) noexcept {
25  if (capacity == 0) {
26  string_buf.reset();
27  tape.reset();

30 
31  // a pathological input like "[[[[..." would generate len tape elements, so
32  // need a capacity of at least len + 1, but it is also possible to do
33  // worse with "[7,7,7,7,6,7,7,7,6,7,7,6,[7,7,7,7,6,7,7,7,6,7,7,6,7,7,7,7,7,7,6"
34  //where len + 1 tape elements are
35  // generated, see issue https://github.com/lemire/simdjson/issues/345
36  size_t tape_capacity = ROUNDUP_N(capacity + 3, 64);
37  // a document with only zero-length strings... could have len/3 string
38  // and we would need len/3 * 5 bytes on the string buffer
39  size_t string_capacity = ROUNDUP_N(5 * capacity / 3 + 32, 64);
40  string_buf.reset( new (std::nothrow) uint8_t[string_capacity]);
41  tape.reset(new (std::nothrow) uint64_t[tape_capacity]);
42  return string_buf && tape ? SUCCESS : MEMALLOC;
43 }

45 inline bool document::dump_raw_tape(std::ostream &os) const noexcept {
46  uint32_t string_length;
47  size_t tape_idx = 0;
48  uint64_t tape_val = tape[tape_idx];
49  uint8_t type = uint8_t(tape_val >> 56);
50  os << tape_idx << " : " << type;
51  tape_idx++;
52  size_t how_many = 0;
53  if (type == 'r') {
54  how_many = size_t(tape_val & internal::JSON_VALUE_MASK);
55  } else {
56  // Error: no starting root node?
57  return false;
58  }
59  os << "\t// pointing to " << how_many << " (right after last node)\n";
60  uint64_t payload;
61  for (; tape_idx < how_many; tape_idx++) {
62  os << tape_idx << " : ";
63  tape_val = tape[tape_idx];
64  payload = tape_val & internal::JSON_VALUE_MASK;
65  type = uint8_t(tape_val >> 56);
66  switch (type) {
67  case '"': // we have a string
68  os << "string \"";
69  memcpy(&string_length, string_buf.get() + payload, sizeof(uint32_t));
70  os << internal::escape_json_string(std::string_view(
71  (const char *)(string_buf.get() + payload + sizeof(uint32_t)),
72  string_length
73  ));
74  os << '"';
75  os << '\n';
76  break;
77  case 'l': // we have a long int
78  if (tape_idx + 1 >= how_many) {
79  return false;
80  }
81  os << "integer " << static_cast<int64_t>(tape[++tape_idx]) << "\n";
82  break;
83  case 'u': // we have a long uint
84  if (tape_idx + 1 >= how_many) {
85  return false;
86  }
87  os << "unsigned integer " << tape[++tape_idx] << "\n";
88  break;
89  case 'd': // we have a double
90  os << "float ";
91  if (tape_idx + 1 >= how_many) {
92  return false;
93  }
94  double answer;
95  memcpy(&answer, &tape[++tape_idx], sizeof(answer));
96  os << answer << '\n';
97  break;
98  case 'n': // we have a null
99  os << "null\n";
100  break;
101  case 't': // we have a true
102  os << "true\n";
103  break;
104  case 'f': // we have a false
105  os << "false\n";
106  break;
107  case '{': // we have an object
108  os << "{\t// pointing to next tape location " << payload
109  << " (first node after the scope) \n";
110  break;
111  case '}': // we end an object
112  os << "}\t// pointing to previous tape location " << payload
113  << " (start of the scope) \n";
114  break;
115  case '[': // we start an array
116  os << "[\t// pointing to next tape location " << payload
117  << " (first node after the scope) \n";
118  break;
119  case ']': // we end an array
120  os << "]\t// pointing to previous tape location " << payload
121  << " (start of the scope) \n";
122  break;
123  case 'r': // we start and end with the root node
124  // should we be hitting the root node?
125  return false;
126  default:
127  return false;
128  }
129  }
130  tape_val = tape[tape_idx];
131  payload = tape_val & internal::JSON_VALUE_MASK;
132  type = uint8_t(tape_val >> 56);
133  os << tape_idx << " : " << type << "\t// pointing to " << payload
134  << " (start root)\n";
135  return true;
136 }
